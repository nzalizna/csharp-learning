// МАСИВИ — ШПАРГАЛКА

// Оголошення і створення
// Порожній масив на 10 місць:
double[] oceny = new double[10];

// Масив з даними відразу:
double[] oceny = new double[] { 4, 3, 5, 3, 4 };

// Короткий запис (те саме):
double[] oceny = { 4, 3, 5, 3, 4 };

// Доступ до елементів:
oceny[0]  // = 4 (перший, індекс завжди з 0!)
oceny[1]  // = 3 (другий)
oceny[4]  // = 4 (п'ятий, останній)
oceny[oceny.Length - 1]  // = 4 (останній, універсально)

// Кількість елементів:
oceny.Length  // = 5

// ВАЖЛИВО: розмір не можна змінити після створення!

// ═══════════════════════════════

// КОЛЕКЦІЇ (List) — ШПАРГАЛКА

List<string> nazwiska = new List<string>();

nazwiska.Add("Kowalski");     // додає в кінець
nazwiska.Add("Nowak");
nazwiska.Remove("Kowalski");  // видаляє (якщо немає — нічого)
nazwiska[0]                   // = "Nowak"
nazwiska.Count                // = 1

// Різниця масив vs List:
// Масив → фіксований розмір → .Length
// List  → змінний розмір   → .Count

// ═══════════════════════════════

// ЦИКЛ FOR — ШПАРГАЛКА

double[] oceny = { 4, 3, 5, 3, 4 };
double suma = 0;

for (int i = 0; i < oceny.Length; i++)
{
    suma += oceny[i];
}
// Крок 1: i=0 → suma = 0 + 4 = 4
// Крок 2: i=1 → suma = 4 + 3 = 7
// Крок 3: i=2 → suma = 7 + 5 = 12
// Крок 4: i=3 → suma = 12 + 3 = 15
// Крок 5: i=4 → suma = 15 + 4 = 19

// Три частини for:
// int i = 0        → починаємо з 0
// i < oceny.Length → допоки не кінець масиву
// i++              → переходимо до наступного

// ═══════════════════════════════

// СКОРОЧЕНІ ОПЕРАТОРИ

i++      // i = i + 1  (наступний крок)
i--      // i = i - 1  (попередній крок)
x += 5   // x = x + 5  (додай 5)
x -= 5   // x = x - 5  (відніми 5)
x *= 5   // x = x * 5  (помнож на 5)
x /= 5   // x = x / 5  (розділи на 5)


// ═══════════════════════════════

// ЦИКЛИ — КОЛИ ЯКИЙ ВИКОРИСТОВУВАТИ

// foreach — найпростіший, коли просто проходиш по всіх елементах
foreach (double ocena in oceny)
{
    suma += ocena;  // ocena = поточний елемент автоматично
}

// for — коли потрібен індекс i
for (int i = 0; i < oceny.Length; i++)
{
    suma += oceny[i];  // oceny[i] = елемент по індексу
}

// while — коли не знаєш скільки разів повторювати
int licznik = 0;
while (licznik < oceny.Length)
{
    suma += oceny[licznik];
    licznik++;  // ВАЖЛИВО: без цього — нескінченний цикл!
}

// do-while — виконується ЗАВЖДИ хоча б один раз
do
{
    suma += oceny[licznik];
    licznik++;
} while (licznik < oceny.Length);  // крапка з комою обов'язкова!

// ═══════════════════════════════

// ПІДСУМОК: ЯКИЙ ЦИКЛ ОБРАТИ?

// foreach  → просто пройти по всіх елементах
// for      → потрібен індекс або не всі елементи
// while    → не знаєш скільки разів
// do-while → треба виконати хоча б раз (рідко)
